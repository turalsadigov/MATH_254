5 not normal, what to do?

Permutation Test!

set.seed(987)
perm_dist <- 
  mice %>% 
  specify(survival_days~group) %>% 
  hypothesize(null = 'independence') %>% 
  generate(reps = 1000, type = 'permute') %>% 
  calculate(stat = 'diff in means', 
            order = c("T", "C")) 

perm_dist %>% 
  visualize()

obs_stat <- 
  mice %>% 
  specify(survival_days~group) %>% 
  calculate(stat = 'diff in means',
            order = c("T", "C"))
