Bootstrap samples

set.seed(49)
treatment_boot <- 
  treatment %>% 
  rep_sample_n(size = 7, 
               replace = TRUE, 
               reps = 1000) %>% 
  summarise(x_bars = mean(x))

set.seed(314)
control_boot <- 
  control %>% 
  rep_sample_n(size = 9, 
               replace = TRUE, 
               reps = 1000) %>% 
  summarise(y_bars = mean(y))
