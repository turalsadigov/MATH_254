5 Bootstrap distribution of sample median

Collapse/show the code below 
set.seed(1234)
delays_sample %>% 
  rep_sample_n(size = 100, 
               replace = T, 
               reps = 1000) %>% 
  group_by(replicate) %>% 
  summarise(meds = median(`Departure Delay Minutes`)) %>% 
  ggplot(aes(meds)) +
  geom_density(fill = 'blue', alpha = 0.4)

6 99% Bootstrap Percentile Confidence Interval

Collapse/show the code below 
set.seed(1234)
delays_sample %>% 
  rep_sample_n(size = 100, 
               replace = T, 
               reps = 1000) %>% 
  group_by(replicate) %>% 
  summarise(meds = median(`Departure Delay Minutes`)) %>% 
  summarise(ave_medians = mean(meds), 
            left = quantile(meds, 0.005),
            right = quantile(meds, 0.995))
